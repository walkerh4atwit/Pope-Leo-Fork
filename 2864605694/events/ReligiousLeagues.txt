# Victory of the Catholic League
country_event = {
	id = religious_leagues.3
	title = "religious_leagues.3.t"
	desc = "religious_leagues.3.d"
	picture = RELIGIOUS_WARS_eventPicture
	major = yes

	is_triggered_only = yes
	
	trigger = {
		has_dlc = "Art of War"
		OR = {
			hre_religion = orthodox
			hre_religion = protestant
			hre_religion = reformed
			hre_religion = anglican
			hre_religion = hussite
			hre_religion = zwinglist
			hre_religion = knoxite
			hre_religion = vasantine
			hre_religion = tausenite
			hre_religion = huguenot
			hre_religion = julianic
			hre_religion = arminian
			hre_religion = socinian
			hre_religion = bironic
			hre_religion = radzian
			hre_religion = molinist
			hre_religion = hutterite
			hre_religion = munsterite
			hre_religion = mennonite
		}
		OR = {
			NOT = { has_country_flag = hre_religion_changed }
			had_country_flag = { flag = hre_religion_changed days = 30 }
		}
	}
	
	immediate = {
		set_country_flag = hre_religion_changed
		hidden_effect = {
			set_hre_heretic_religion = protestant
			set_hre_religion = catholic
			set_hre_religion_locked = yes
		}
	}

	option = {
		name = "religious_leagues.3.a"
		tooltip = { 
			set_hre_heretic_religion = protestant
			set_hre_religion = catholic
			set_hre_religion_locked = yes
		}
	}
}

	title = "religious_leagues.29.t"
	desc = "religious_leagues.29.d"
	picture = RELIGIOUS_WARS_eventPicture
	major = yes

	is_triggered_only = yes
	
	trigger = {
		has_dlc = "Art of War"
		hre_religion = catholic
		hre_heretic_religion = anglican
		OR = {
			NOT = { has_country_flag = hre_religion_changed }
			had_country_flag = { flag = hre_religion_changed days = 30 }
		}
	}
	
	immediate = {
		set_country_flag = hre_religion_changed
		hidden_effect = {
			#set to treaty first so we can set everything else without elections being called at the wrong time
			set_hre_religion_treaty = yes
			set_hre_heretic_religion = catholic
			set_hre_religion = anglican

			#now set it to locked once we've completed the tasks
			set_hre_religion_locked = yes
		}
	}

	option = {
		name = "religious_leagues.29.a"
		tooltip = { 
			set_hre_heretic_religion = catholic
			set_hre_religion = anglican
			set_hre_religion_locked = yes			
		}

		# this should elect a new emperor
		set_hre_religion = anglican
	}
}

# The Imperial Diet (Catholic)
country_event = {
	id = religious_leagues.6
	title = "religious_leagues.6.t"
	desc = "religious_leagues.6.d"
	picture = RELIGIOUS_WARS_eventPicture
	major = yes
	fire_only_once = yes
	
	trigger = {
		has_dlc = "Art of War"
		is_emperor = yes
		is_at_war = no
		NOT = {
			any_known_country = {
				is_in_league_war = yes
			}
		}
		has_regency = no
		hre_religion_locked = no
		hre_religion_treaty = no
		religion = catholic
		OR = {
			AND = {
				hre_leagues_enabled = yes
				had_global_flag = {
					flag = evangelical_union_happened
					days = 10950
				}		
				NOT = {
					any_known_country = {
						is_elector = yes
						OR = {
							religion = catholic
							religion = orthodox
							religion = protestant
							religion = reformed
							religion = anglican
							religion = hussite
							religion = zwinglist
							religion = knoxite
							religion = vasantine
							religion = tausenite
							religion = huguenot
							religion = julianic
							religion = arminian
							religion = socinian
							religion = bironic
							religion = radzian
							religion = molinist
							religion = hutterite
							religion = munsterite
							religion = mennonite
						}
						is_league_enemy = ROOT
						truce_with = ROOT
					}		
				}
			}
			AND = {
				NOT = { has_global_flag = evangelical_union_happened }
				is_year = 1625
			}
		}
	}
	
	mean_time_to_happen = {
		months = 60
	}

	option = {
		name = "religious_leagues.6.a"
		set_hre_religion_locked = yes
	}
}
