#Theologians Debate Our Beliefs (Use for All Branches)
country_event = {
	id = mb.91
	title = "mb.91.t"
	desc = "mb.91.d"
	picture = PROTESTANT_CATHOLIC_DEBATE_eventPicture
	
	trigger = {
		OR = {
			religion = zwinglist
			religion = knoxite
			religion = vasantine
			religion = tausenite
			religion = huguenot
			religion = julianic
			religion = arminian
			religion = socinian
			religion = bironic
			religion = radzian
			religion = molinist
			religion = hutterite
			religion = munsterite
			religion = mennonite
		}
		if = { #Uses Church Power
			limit = {
				OR = {
					religion = zwinglist
					religion = tausenite
					religion = arminian
					religion = bironic
					religion = radzian
					religion = molinist
				}
			}
			NOT = { num_of_aspects = 3 }
			NOT = { church_power = 200 }
		}
		if = { #Uses Fervor
			limit = {
				OR = {
					religion = knoxite
					religion = vasantine
					religion = huguenot
					religion = julianic
					religion = socinian
					religion = hutterite
					religion = munsterite
					religion = mennonite
				}
			}
			uses_fervor = yes
		}
		NOT = { has_country_flag = mb_ongoing_debate }
	}

	mean_time_to_happen = {
		months = 1440
	}
	
	option = { #Encourage Debate
		name = "mb.91.a"
		set_country_flag = mb_ongoing_debate
		add_dip_power = -100
		add_country_modifier = {
			name = mb_ongoing_debate
			duration = 9125
		}
	}
	option = { #Discourage Debate
		name = "mb.91.b"
		set_country_flag = mb_ongoing_debate
		if = {
			limit = {
				OR = { #Uses Church Power
					religion = zwinglist
					religion = tausenite
					religion = arminian
					religion = bironic
					religion = radzian
					religion = molinist
				}
			}
			add_church_power = -25
		}
		if = {
			limit = { #Uses Fervor
				OR = {
					religion = knoxite
					religion = vasantine
					religion = huguenot
					religion = julianic
					religion = socinian
					religion = hutterite
					religion = munsterite
					religion = mennonite
				}
			}
			add_fervor = -10
		}
	}
}

#Religious Fanatics (zwinglist)
province_event = {
	id = mb.79
	title = "mb.79.t"
	desc = "mb.79.d"
	picture = RELIGIOUS_TURMOIL_eventPicture
	
	trigger = {
		owner = {
			religion = zwinglist
			NOT = { num_of_aspects = 3 }
			NOT = { church_power = 200 }
			#uses_fervor = yes
		}
		NOT = { religion = zwinglist }
		NOT = { has_province_flag = mb_fanatics }
		NOT = { has_province_flag = mb_stopped_fanatics }
	}

	mean_time_to_happen = {
		months = 1800
		modifier = {
			factor = 0.80
			any_neighbor_province = {
				owned_by = ROOT
				has_province_flag = mb_fanatics
			}
		}
		modifier = {
			factor = 1.2
			any_neighbor_province = {
				owned_by = ROOT
				has_province_flag = mb_stopped_fanatics
			}
		}
	}
	
	option = { #Stop the fanatics
		name = "mb.79.a"
		set_province_flag = mb_stopped_fanatics
		owner = {
			#add_fervor = -10
			add_church_power = -20
			random_owned_province = {
				limit = {
					continent = europe
					religion = zwinglist
				}
				add_unrest = 1
			}
		}
	}
	option = { #Encourage fanatics
		name = "mb.79.b"
		set_province_flag = mb_fanatics
		owner = {
			#add_fervor = 10
			add_church_power = 25
		}
		spawn_rebels = {
			type = zwinglist_rebels
			size = 2
		}
		add_province_modifier = {
			name = religious_violence
			duration = 7300
		}
	}
}
